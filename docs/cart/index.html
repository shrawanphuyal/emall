# Checkout

METHODPOST
URLhttp://emall.com.np/api/checkout
PARAMS
  • [String] phone*
  • [String] address*
  • [Array] cartItems*
AUTHtrue
HEADERS
  • Accept: application/json
  • Authorization: Bearer {token}

# “cartItems” format must be as below

{
	"cartItems": [
		{
			"id": 1,
			"quantity": 4,
			"size": 4
		},
		{
			"id": 1,
			"quantity": 4,
			"size": 0 // if there is no size, send 0
		}
	]
}

# Success Response

{
	"status": true,
	"message": "Order successfully placed. We will contact you soon.",
	"code": 200
}
